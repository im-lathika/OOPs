import java.util.*;

class ProductNotFoundException extends Exception {
    public ProductNotFoundException(String msg) {
        super(msg);
    }
}

class InvalidQuantityException extends Exception {
    public InvalidQuantityException(String msg) {
        super(msg);
    }
}

class EmptyCartException extends Exception {
    public EmptyCartException(String msg) {
        super(msg);
    }
}

class Product<T> {
    private T id;
    private String name;
    private double price;

    public Product(T id, String name, double price) {
        this.id = id;
        this.name = name;
        this.price = price;
    }

    public T getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public double getPrice() {
        return price;
    }
}

class Cart {
    private List<Product<?>> products = new ArrayList<>();

    public void addProduct(Product<?> p) {
        products.add(p);
    }

    public void removeProduct(int index) throws ProductNotFoundException {
        if (index < 0 || index >= products.size()) {
            throw new ProductNotFoundException("Invalid product index!");
        }
        products.remove(index);
    }

    public double calculateTotal() throws EmptyCartException {
        if (products.isEmpty()) {
            throw new EmptyCartException("Cart is empty!");
        }

        double total = 0;

        for (Product<?> p : products) {
            total += p.getPrice();
        }

        return total;
    }

    public void displayCart() {
        if (products.isEmpty()) {
            System.out.println("Cart is empty!");
            return;
        }

        System.out.println("\nCart Items:");

        int i = 0;
        for (Product<?> p : products) {
            System.out.println(i + " - " + p.getName() + " : ₹" + p.getPrice());
            i++;
        }
    }
}

class BillingService {

    public void generateBill(Cart cart) {
        try {
            double total = cart.calculateTotal();
            double discount = applyDiscount(total);
            double finalAmount = total - discount;

            System.out.println("\n------ BILL ------");
            System.out.println("Total Amount: ₹" + total);
            System.out.println("Discount: ₹" + discount);
            System.out.println("Final Amount: ₹" + finalAmount);
            System.out.println("------------------");

        } catch (EmptyCartException e) {
            System.out.println(e.getMessage());
        }
    }

    private double applyDiscount(double total) {
        if (total > 1000) {
            return total * 0.10;
        }
        return 0;
    }
}

public class BillingSystem {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        Map<Integer, Product<Integer>> store = new HashMap<>();

        store.put(1, new Product<>(1, "Pen", 10));
        store.put(2, new Product<>(2, "Book", 50));
        store.put(3, new Product<>(3, "Bag", 500));
        store.put(4, new Product<>(4, "Shoes", 1200));

        Cart cart = new Cart();
        BillingService billing = new BillingService();

        while (true) {

            System.out.println("\n1. View Products");
            System.out.println("2. Add to Cart");
            System.out.println("3. Remove from Cart");
            System.out.println("4. View Cart");
            System.out.println("5. Generate Bill");
            System.out.println("6. Exit");

            System.out.print("Enter choice: ");
            int choice = sc.nextInt();

            if (choice == 1) {

                System.out.println("\nAvailable Products:");

                for (Product<Integer> p : store.values()) {
                    System.out.println(
                            p.getId() + " - " + p.getName() + " : ₹" + p.getPrice());
                }

            } else if (choice == 2) {

                System.out.print("Enter Product ID: ");
                int id = sc.nextInt();

                try {
                    if (!store.containsKey(id)) {
                        throw new ProductNotFoundException("Product not found!");
                    }

                    cart.addProduct(store.get(id));
                    System.out.println("Product added to cart!");

                } catch (ProductNotFoundException e) {
                    System.out.println(e.getMessage());
                }

            } else if (choice == 3) {

                cart.displayCart();

                System.out.print("Enter index to remove: ");
                int index = sc.nextInt();

                try {
                    cart.removeProduct(index);
                    System.out.println("Product removed!");

                } catch (ProductNotFoundException e) {
                    System.out.println(e.getMessage());
                }

            } else if (choice == 4) {

                cart.displayCart();

            } else if (choice == 5) {

                billing.generateBill(cart);

            } else if (choice == 6) {

                System.out.println("Thank you!");
                sc.close();
                System.exit(0);

            } else {

                System.out.println("Invalid choice!");
            }
        }
    }
}
